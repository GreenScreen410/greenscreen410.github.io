## 문제

<https://www.acmicpc.net/problem/23802>

## 풀이

단순 골뱅이 찍기 문제입니다.

```cpp
#include <iostream>
using namespace std;

int main() {
  int N;
  cin >> N;

  for (int i = 0; i < N; i++) {
    for (int j = 0; j < N * 5; j++) {
      cout << "@";
    }
    cout << "\n";
  }
  for (int i = 0; i < N; i++) {
    for (int j = 0; j < N; j++) {
      cout << "@";
    }
    cout << "\n";
  }
  for (int i = 0; i < N; i++) {
    for (int j = 0; j < N * 5; j++) {
      cout << "@";
    }
    cout << "\n";
  }
  for (int i = 0; i < N * 2; i++) {
    for (int j = 0; j < N; j++) {
      cout << "@";
    }
    cout << "\n";
  }

  return 0;
}
```
