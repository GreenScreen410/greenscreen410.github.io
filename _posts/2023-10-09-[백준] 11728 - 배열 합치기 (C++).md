## 문제

<https://www.acmicpc.net/problem/11728>

## 풀이

C 배열로 합치기 위하여 insert 함수를 사용하였습니다.

```cpp
#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;

int main() {
  int N, M;
  cin >> N >> M;

  vector<int> A(N);
  for (int i = 0; i < N; i++) {
    cin >> A[i];
  }

  vector<int> B(M);
  for (int i = 0; i < M; i++) {
    cin >> B[i];
  }

  vector<int> C;
  C.insert(C.begin(), A.begin(), A.end());
  C.insert(C.begin(), B.begin(), B.end());
  sort(C.begin(), C.end());

  for (int i = 0; i < C.size(); i++) {
    cout << C[i] << " ";
  }

  return 0;
}
```
